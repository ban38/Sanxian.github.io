C++解題

1. 輸入一個數字，判斷是否為質數。

這個程式碼會先讓使用者輸入一個數字，然後判斷是否為質數。判斷質數的過程是，先將數字除以2到數字的平方根，如果有餘數，則為非質數。如果沒有餘數，則為質數。

                #include <iostream>
                using namespace std;
                int main() 
                {
                    int num;
                    cout << "Enter a number: ";
                    cin >> num;
                    bool prime = true;
                    for (int i = 2; i < num; i++) 
                    {
                        if (num % i == 0) 
                        {
                            prime = false;
                            break;
                        }
                    }
                    if (prime) 
                    {
                        cout << num << " is a prime number." << endl;
                    } 
                    else 
                    {
                        cout << num << " is not a prime number." << endl;
                    }
                    return 0;
                }

2. 輸入一個數字，判斷是否為負數。

這個程式碼會先讓使用者輸入一個數字，然後判斷是否為負數。判斷負數的過程是，只要數字小於0，就表示為負數。

                #include <iostream>
                using namespace std;

                int main() 
                {
                    int num;
                    cout << "Enter a number: ";
                    cin >> num;
                    if (num < 0)
                    {
                        cout << num << " is a negative number." << endl;
                    } 
                    else 
                    {
                        cout << num << " is not a negative number." << endl;
                    }
                    return 0;
                }

3. 輸入一個數字，判斷是否為奇數。

這個程式碼會先讓使用者輸入一個數字，然後判斷是否為奇數。判斷奇數的過程是，只要數字除以2餘1，就表示為奇數。

                #include <iostream>
                using namespace std;

                int main() 
                {
                    int num;
                    cout << "Enter a number: ";
                    cin >> num;
                    if (num % 2 == 1) 
                    {
                        cout << num << " is an odd number." << endl;
                    } 
                    else 
                    {
                        cout << num << " is an even number." << endl;
                    }
                    return 0;
                }

4. 輸入兩個數字，計算其乘積。

這個程式碼會先讓使用者輸入兩個數字，然後計算其乘積。計算乘積的過程是，將兩個數字相乘。

                #include <iostream>
                using namespace std;

                int main() 
                {
                    int num1, num2;
                    cout << "Enter two numbers: ";
                    cin >> num1 >> num2;
                    int product = num1 * num2;
                    cout << "The product of " << num1 << " and " << num2 << " is " << product << "." << endl;
                    return 0;
                }

5. 輸入一個數字，計算其平方根。

這個程式碼會先讓使用者輸入一個數字，然後計算其平方根。計算平方根的過程是，使用cmath中的sqrt()函數。

                #include <iostream>
                #include <cmath>
                using namespace std;

                int main() 
                {
                    int num;
                    cout << "Enter a number: ";
                    cin >> num;
                    double sqrt_num = sqrt(num);
                    cout << "The square root of " << num << " is " << sqrt_num << "." << endl;
                    return 0;
                }
